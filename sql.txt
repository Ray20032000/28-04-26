CREATE TABLE usuario (
id_usuario INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR(100) NOT NULL,
email VARCHAR(100) NOT NULL,
telefone VARCHAR(20) NOT NULL,
senha VARCHAR(20) NOT NULL,
cargo INT NOT NULL 
)


CREATE TABLE chamado (
id_chamado INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
sala VARCHAR(20) NOT NULL,
patrimonio VARCHAR(50) NOT NULL,
descricao BLOB SUB_TYPE TEXT NOT NULL,
data_abertura TIMESTAMP NOT NULL,
concluido TIMESTAMP NOT NULL,
id_status INT NOT NULL,
id_usuario INT NOT NULL,

    FOREIGN KEY (id_status) REFERENCES status(id_status),
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario)
)

CREATE TABLE comentario (
id_comentario INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
texto BLOB SUB_TYPE TEXT NOT NULL,
data TIMESTAMP NOT NULL,
id_chamado INT,
id_usuario INT,

    FOREIGN KEY (id_chamado) REFERENCES chamado(id_chamado),
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario)
)

CREATE TABLE status (
id_status INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR(30) NOT NULL
)
